### Divided Korea: NKO is released

country_event = {
	id = 1209002

	title = EVT_1209002_NAME
	desc = EVT_1209002_DESC
	picture = "divided_korea"

	fire_only_once = yes

	trigger = {
		tag = RUS
		government = proletarian_dictatorship
		NOT = { exists = NKO }
		1619 = { controlled_by = RUS } # Pyongyang
		OR = { 
			year = 1947
			has_country_flag = korea_trusteeship
		}
	}

	mean_time_to_happen = {
		months = 22
		modifier = {
			factor = 0.1
			SKO = { exists = yes } 
		}
	}

	option = {
		name = EVT_1209002_A
		ai_chance = { factor = 100 }
		release_vassal = NKO
		diplomatic_influence = { who = NKO value = 100 }
		set_country_flag = korea_trusteeship
	}
	
	option = {
		name = EVT_1209002_B # We will remain there
		ai_chance = { factor = 0 }
		set_country_flag = korea_trusteeship
		any_pop = {
			limit = {
				culture = korean
				is_primary_culture = no
				is_accepted_culture = no
			}
			militancy = 10
			consciousness = 4
	   }
	}

}
