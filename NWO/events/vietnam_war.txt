# South Vietnam 
country_event = {
	id = 1228001

	title = EVT_1228001_NAME
	desc = EVT_1228001_DESC
	picture = "french_vietnam"

	fire_only_once = yes
	
	trigger = {
		tag = FRA
		NOT = { exists = SVI }
		exists = DAI
		war = no
		1380 = { controlled_by = FRA } # Saigon
		1369 = { NOT = { controlled_by = FRA } } # Hanoi
	}

	mean_time_to_happen = {
		months = 2
	}

	option = {
		name = EVT_1228001_A # South Vietnam Independence
		ai_chance = { factor = 0 }
		prestige = -20 
		release = SVI
		release_vassal = CAM diplomatic_influence = { who = CAM value = -100 }
		release_vassal = LUA diplomatic_influence = { who = LUA value = -100 }
	}
	
	option = {
		name = EVT_1228001_B # Leave Vietnam to it's faith
		ai_chance = { factor = 100 }
		USA = { country_event = 1228002 }
		release_vassal = CAM diplomatic_influence = { who = CAM value = -100 }
		release_vassal = LUA diplomatic_influence = { who = LUA value = -100 }
	}

	option = {
		name = EVT_1228001_C
		ai_chance = { factor = 0 }
		any_pop = {
			limit = {
				culture = vietnamese
				is_primary_culture = no
				is_accepted_culture = no
			}
			militancy = 10
	   }
	   release_vassal = SVI
	}

}
